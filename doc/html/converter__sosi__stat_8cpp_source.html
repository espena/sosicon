77 run() {

79  std::cout << "\nGenerating statistics for " << *f << "\n";
80  Parser p;
81  char ln[ 1024 ];

87  std::cout << "\rParsing " << c << " lines...";

93  std::cout << "\n" << c << " lines in file \n\n";

95  ifs.close();
96

102  sosi::SosiElementSearch srcHeadItem;
103
